cat /etc/ssh/sshd_config

DESCOMENTE A LINHA ABAIXO

VERIFICA O STATUS DO SERVIÇO SSHD, DÁ PARA VERIFICAR QUEM TENTOU ACESSAR O SERVIDOR VIA SSH
systemctl status sshd

O SERVIÇO SSHD PRECISA SER REINICIADO PARA ACEITAR AS ALTERAÇÕES NO ARQUIVO
systemctl restart sshd
